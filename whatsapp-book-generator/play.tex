% Create chat bubble environments
\newenvironment{chatbubble}[2]{%
    \par\noindent

    \def\content{#1}
    \settowidth{\mylen}{\content}

    \ifdim\mylen<.8\textwidth

    \else
    \begin{tcolorbox}[
        arc=2mm,
        boxrule=0pt,
        boxsep=0pt,
        left=2mm,
        right=2mm,
        top=2mm,
        bottom=2mm,
        colback=#1,
        coltext=#2,
        width=0.7\textwidth,
        before=\ifstrequal{#1}{bluebubble}{\hfill}{},
    ]
}{%
    \end{tcolorbox}
    \vspace{-1mm}
}
